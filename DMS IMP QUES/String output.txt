#include <stdio.h>
int main(){
    int i,j;
    char a[100];
    gets(a);
    for(i=0;a[i]!='\0';i++);
    int b[i-1],n=i-1;
    for(i=0;i<=n;i++){
        int count=0;
        for(j=0;j<=n;j++){
            if(a[i]==a[j])
                count++;
            b[i]=count;
        }
    }
    int max=b[0];
    for(i=0;i<=n;i++){
        if(b[i]>max)
            max=b[i];
        for(j=0;j<=n;j++){
            if(i!=j && j>i && a[i]==a[j])
                b[j]=0;
        }
        
    }
    for(i=0;i<=n;i++){
        if(b[i]==max)
            printf("%c",a[i]);
    }
    return 0;
}
